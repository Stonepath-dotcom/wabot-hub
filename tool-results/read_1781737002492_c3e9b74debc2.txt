    73тЖТ    73тЖТfunction fileToDataUrl(file: File): Promise<string> {
    74тЖТ    74тЖТ  return new Promise((resolve, reject) => {
    75тЖТ    75тЖТ    const reader = new FileReader()
    76тЖТ    76тЖТ    reader.onload = () => resolve(reader.result as string)
    77тЖТ    77тЖТ    reader.onerror = reject
    78тЖТ    78тЖТ    reader.readAsDataURL(file)
    79тЖТ    79тЖТ  })
    80тЖТ    80тЖТ}

    82тЖТ    82тЖТfunction imageToJpegBytes(dataUrl: string): Promise<Uint8Array> {
    83тЖТ    83тЖТ  return new Promise((resolve, reject) => {
    84тЖТ    84тЖТ    const img = new Image()
    85тЖТ    85тЖТ    img.onload = () => {
    86тЖТ    86тЖТ      const canvas = document.createElement('canvas')
    87тЖТ    87тЖТ      canvas.width = img.naturalWidth
    88тЖТ    88тЖТ      canvas.height = img.naturalHeight
    89тЖТ    89тЖТ      const ctx = canvas.getContext('2d')!
    90тЖТ    90тЖТ      ctx.drawImage(img, 0, 0)
    91тЖТ    91тЖТ      canvas.toBlob(
    92тЖТ    92тЖТ        (blob) => {
    93тЖТ    93тЖТ          if (!blob) { reject(new Error('Canvas toBlob gagal')); return }
    94тЖТ    94тЖТ          blob.arrayBuffer().then((buf) => resolve(new Uint8Array(buf))).catch(reject)
    95тЖТ    95тЖТ        },
    96тЖТ    96тЖТ        'image/jpeg',
    97тЖТ    97тЖТ        0.92
    98тЖТ    98тЖТ      )
    99тЖТ    99тЖТ    }
   100тЖТ   100тЖТ    img.onerror = () => reject(new Error('Gambar tidak bisa dibaca'))
   101тЖТ   101тЖТ    img.src = dataUrl
   102тЖТ   102тЖТ  })
   103тЖТ   103тЖТ}

   105тЖТ   105тЖТasync function addImageToPdf(pdfDoc: PDFDocument, dataUrl: string) {
   106тЖТ   106тЖТ  const jpgBytes = await imageToJpegBytes(dataUrl)
   107тЖТ   107тЖТ  const image = await pdfDoc.embedJpg(jpgBytes)
   108тЖТ   108тЖТ
   109тЖТ   109тЖТ  const page = pdfDoc.addPage([595, 842]) // A4
   110тЖТ   110тЖТ  const pw = 595
   111тЖТ   111тЖТ  const ph = 842
   112тЖТ   112тЖТ  const { width: iw, height: ih } = image.scale(1)
   113тЖТ   113тЖТ
   114тЖТ   114тЖТ  const scale = Math.min(pw / iw, ph / ih)
   115тЖТ   115тЖТ  const drawW = iw * scale
   116тЖТ   116тЖТ  const drawH = ih * scale
   117тЖТ   117тЖТ  const x = (pw - drawW) / 2
   118тЖТ   118тЖТ  const y = (ph - drawH) / 2
   119тЖТ   119тЖТ
   120тЖТ   120тЖТ  page.drawImage(image, { x, y, width: drawW, height: drawH })
   121тЖТ   121тЖТ}
